export const ThemeStorage = {
  async save(preferences: ThemePreferences): Promise<void> {
    try {
      await AsyncStorage.setItem(
        THEME_STORAGE_KEY, 
        JSON.stringify(preferences)
      );

      logger.error('Theme preferences load failed:', error);

    }

};
